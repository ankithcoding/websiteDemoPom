ALTER TABLE `Licence`   
  ADD COLUMN `OWNER_COMPANY_ID` BIGINT(20) NULL;

UPDATE Licence l
LEFT JOIN Product p ON
    l.PRODUCT_ID = p.id
SET
    l.OWNER_COMPANY_ID = p.OWNER_COMPANY_ID;  
  
    
SELECT l.PRODUCT_ID, l.OWNER_COMPANY_ID, p.OWNER_COMPANY_ID FROM Licence l LEFT OUTER JOIN Product p  ON (l.PRODUCT_ID = p.id);
